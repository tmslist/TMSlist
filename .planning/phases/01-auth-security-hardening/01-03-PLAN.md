/**
 * Check if user has a specific granular permission.
 * Falls back to checking role if permissions field is absent (backwards compat).
 * Admins with role='admin' get all permissions unless explicitly overridden.
 */
export function hasPermission(session: JWTPayload | null, permission: keyof UserPermissions): boolean {
  if (!session) return false;

  // Admins get all permissions by default
  if (session.role === 'admin') return true;

  // For non-admins, check the permissions JSONB field from the session payload
  // Note: JWT payload doesn't include permissions by default — we fetch from DB in API routes
  // This function is called from API routes after fetching user permissions
  // For role-based checks that can be done from JWT alone, use hasRole instead
  return false;
}

/**
 * Parse session expiry string to milliseconds.
 * Options: '1h', '8h', '24h', '30d'
 */
export function parseSessionExpiry(expiry: string | null | undefined): number {
  switch (expiry) {
    case '1h': return 1 * 60 * 60 * 1000;
    case '8h': return 8 * 60 * 60 * 1000;
    case '24h': return 24 * 60 * 60 * 1000;
    case '30d': return 30 * 24 * 60 * 60 * 1000;
    default: return 8 * 60 * 60 * 1000; // default to 8h
  }
}

/**
 * Create a session with configurable expiry (for remember-me flows).
 * Replaces createSession when rememberMe is needed.
 */
export async function createSessionWithExpiry(
  payload: JWTPayload,
  opts?: {
    userAgent?: string;
    ipAddress?: string;
    rememberMe?: boolean;
    sessionExpiryDays?: string;
  }
): Promise<{ cookie: string; sessionId: string }> {
  // Determine expiry based on user preference and settings
  let expiryMs: number;
  if (opts?.rememberMe) {
    expiryMs = parseSessionExpiry('30d');
  } else if (opts?.sessionExpiryDays) {
    expiryMs = parseSessionExpiry(opts.sessionExpiryDays);
  } else {
    expiryMs = parseSessionExpiry('8h'); // default
  }

  const token = createToken(payload);
  const sessionId = randomBytes(16).toString('hex');
  const tokenHash = createHash('sha256').update(token).digest('hex');
  const expiresAt = new Date(Date.now() + expiryMs);

  await db.insert(sessions).values({
    userId: payload.userId,
    tokenHash,
    expiresAt,
    lastUsedAt: new Date(),
    userAgent: opts?.userAgent,
    ipAddress: opts?.ipAddress,
  });

  const secure = (process.env.NODE_ENV === 'production' || process.env.VERCEL === '1')
    ? '; Secure'
    : '';

  // rememberMe = 30 days, otherwise use the configured expiry
  const maxAge = opts?.rememberMe
    ? 30 * 24 * 60 * 60
    : Math.floor(expiryMs / 1000);

  const cookie = `${COOKIE_NAME}=${token}; Path=/; HttpOnly; SameSite=Lax; Max-Age=${maxAge}${secure}`;
  return { cookie, sessionId };
}
```

import { eq } from 'drizzle-orm';

async function requireAdminWithPermission(
  request: Request,
  permission: keyof UserPermissions
): Promise<{ authorized: false; response: Response } | { authorized: true; session: import('../../../utils/auth').JWTPayload; userId: string }> {
  const session = getSessionFromRequest(request);
  if (!session) {
    return {
      authorized: false,
      response: new Response(JSON.stringify({ error: 'Unauthorized' }), {
        status: 401,
        headers: { 'Content-Type': 'application/json' },
      }),
    };
  }

  // Fetch user to check permissions
  const userResults = await db.select({ permissions: users.permissions, role: users.role })
    .from(users)
    .where(eq(users.id, session.userId))
    .limit(1);

  const user = userResults[0];
  if (!user) {
    return {
      authorized: false,
      response: new Response(JSON.stringify({ error: 'Unauthorized' }), {
        status: 401,
        headers: { 'Content-Type': 'application/json' },
      }),
    };
  }

  // Admins with role='admin' have all permissions
  if (user.role === 'admin') return { authorized: true, session, userId: session.userId };

  // Check specific permission
  const perms = user.permissions as UserPermissions | null;
  if (!perms?.[permission]) {
    return {
      authorized: false,
      response: new Response(JSON.stringify({ error: 'Forbidden' }), {
        status: 403,
        headers: { 'Content-Type': 'application/json' },
      }),
    };
  }

  return { authorized: true, session, userId: session.userId };
}
```

/**
 * PUT /api/admin/users/:id/permissions
 * Update permission flags for a user.
 * Requires: admin role OR can_manage_users permission.
 */
export const PUT_permissions: APIRoute = async ({ params, request }) => {
  const auth = await requireAdminWithPermission(request, 'can_manage_users');
  if (!auth.authorized) return auth.response;

  const targetUserId = params.id;
  if (!targetUserId) {
    return new Response(JSON.stringify({ error: 'User ID required' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  if (!/^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(targetUserId)) {
    return new Response(JSON.stringify({ error: 'Invalid user ID format' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  // Prevent self-modification of own permissions
  if (targetUserId === auth.userId) {
    return new Response(JSON.stringify({ error: 'Cannot modify your own permissions' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  const body = await request.json();
  const parsed = z.object({
    can_edit: z.boolean().optional(),
    can_delete: z.boolean().optional(),
    can_export: z.boolean().optional(),
    can_manage_users: z.boolean().optional(),
    can_billing: z.boolean().optional(),
  }).safeParse(body);

  if (!parsed.success) {
    return new Response(JSON.stringify({ error: 'Invalid permissions object' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  // Fetch existing permissions and merge
  const existing = await db.select({ permissions: users.permissions })
    .from(users)
    .where(eq(users.id, targetUserId))
    .limit(1);

  const currentPerms = (existing[0]?.permissions as UserPermissions | null) ?? {
    can_edit: false,
    can_delete: false,
    can_export: false,
    can_manage_users: false,
    can_billing: false,
  };

  const updatedPerms: UserPermissions = {
    ...currentPerms,
    ...parsed.data,
  };

  await db.update(users).set({ permissions: updatedPerms }).where(eq(users.id, targetUserId));

  // Log permission change
  await logAuthEvent({
    userId: auth.userId,
    action: 'permission_changed',
    metadata: { targetUserId, permissions: updatedPerms },
  });

  return new Response(JSON.stringify({ success: true, permissions: updatedPerms }), {
    headers: { 'Content-Type': 'application/json' },
  });
};
```

/**
 * PUT /api/admin/users/:id/session-expiry
 * Update session expiry setting for a user.

export const PUT_sessionExpiry: APIRoute = async ({ params, request }) => {
  const session = getSessionFromRequest(request);
  if (!session) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), {
      status: 401,
      headers: { 'Content-Type': 'application/json' },
    });
  }
  if (!hasRole(session, 'admin')) {
    return new Response(JSON.stringify({ error: 'Forbidden' }), {
      status: 403,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  const targetUserId = params.id;
  if (!targetUserId) {
    return new Response(JSON.stringify({ error: 'User ID required' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  if (!/^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(targetUserId)) {
    return new Response(JSON.stringify({ error: 'Invalid user ID format' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  const body = await request.json();
  const parsed = z.object({
    sessionExpiry: z.enum(['1h', '8h', '24h', '30d']),
  }).safeParse(body);

  if (!parsed.success) {
    return new Response(JSON.stringify({ error: 'sessionExpiry must be one of: 1h, 8h, 24h, 30d' }), {
      status: 400,
      headers: { 'Content-Type': 'application/json' },
    });
  }

  await db.update(users).set({ sessionExpiry: parsed.data.sessionExpiry }).where(eq(users.id, targetUserId));

  return new Response(JSON.stringify({
    success: true,
    sessionExpiry: parsed.data.sessionExpiry,
  }), {
    headers: { 'Content-Type': 'application/json' },
  });
};
```
