1. **`/api/admin/backups` POST (BUG-06a):** The component calls `fetch('/api/admin/backups', { method: 'POST', body: JSON.stringify({}) })` but the component also needs to handle restore. The `restore` endpoint at `/api/admin/backups/restore` needs to be created.

Create `src/pages/api/admin/backups/restore.ts`:
```typescript
import type { APIRoute } from 'astro';
import { eq } from 'drizzle-orm';
import { db } from '../../../../db';
import { backups } from '../../../../db/schema';
import { getSessionFromRequest, hasRole } from '../../../../utils/auth';

export const prerender = false;

export const POST: APIRoute = async ({ request, url }) => {
  const session = getSessionFromRequest(request);
  if (!session || !hasRole(session, 'admin')) {
    return new Response(JSON.stringify({ error: 'Forbidden' }), { status: 403, headers: { 'Content-Type': 'application/json' } });
  }
  const id = url.searchParams.get('id');
  if (!id) return new Response(JSON.stringify({ error: 'id required' }), { status: 400, headers: { 'Content-Type': 'application/json' } });
  try {
    // Update backup status to 'restoring'
    await db.update(backups).set({ metadata: { status: 'restoring', startedAt: new Date().toISOString() } }).where(eq(backups.id, id));
    // In production, this would trigger an async restore job
    return new Response(JSON.stringify({ success: true, message: 'Restore initiated' }), { status: 200, headers: { 'Content-Type': 'application/json' } });
  } catch (err) {
    console.error('Restore error:', err);
    return new Response(JSON.stringify({ error: 'Internal server error' }), { status: 500, headers: { 'Content-Type': 'application/json' } });
  }
};
```

```typescript
export const PUT: APIRoute = async ({ request, url }) => {
  const denied = guard(request); if (denied) return denied;
  const id = url.searchParams.get('id');
  if (!id) return json({ error: 'id required' }, 400);
  try {
    const body = await request.json();
    const [row] = await db.update(backupSchedules).set({
      name: body.name,
      type: body.type,
      frequency: body.frequency,
      retentionCount: body.retentionCount ?? 7,
      enabled: body.enabled ?? true,
    }).where(eq(backupSchedules.id, id)).returning();
    return json({ data: row });
  } catch (err) { console.error('backup-schedules PUT', err); return json({ error: 'Internal server error' }, 500); }
};
```

```typescript
// Replace the in-memory Map with async DB operations
async function saveAudit(url: string, audit: PageAudit) {
  // Store in audit_log with metadata type='seo-audit'
  await db.insert(auditLog).values({
    entityType: 'seo-audit',
    entityId: url,
    action: 'audit_completed',
    metadata: audit,
  }).onConflictDoUpdate({
    target: [auditLog.entityId, auditLog.action],
    set: { metadata: audit, createdAt: new Date() },
  });
}

async function getAudit(url: string): Promise<PageAudit | null> {
  const [result] = await db.select().from(auditLog)
    .where(and(eq(auditLog.entityType, 'seo-audit'), eq(auditLog.entityId, url)))
    .orderBy(desc(auditLog.createdAt)).limit(1);
  return result?.metadata as PageAudit | null;
}
```
