This endpoint reads from the `auth_events` table (created in Phase 1 AUTH-10) and returns recent admin actions for the activity feed.

```typescript
import { json } from '@sveltejs/kit';
import type { RequestHandler } from './$types';
import { db } from '../../../db';
import { authEvents } from '../../../db/schema';
import { desc, eq, and, isNotNull, sql } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../utils/auth';

export const GET: RequestHandler = async ({ request }) => {
  // Auth check
  const session = await getSessionFromRequest(request);
  if (!session || !hasRole(session, 'admin')) {
    return json({ error: 'Unauthorized' }, { status: 401 });
  }

  // Fetch recent auth events (last 24 hours)
  const oneDayAgo = new Date();
  oneDayAgo.setDate(oneDayAgo.getDate() - 1);

  const events = await db
    .select({
      id: authEvents.id,
      action: authEvents.action,
      userId: authEvents.userId,
      ipAddress: authEvents.ipAddress,
      metadata: authEvents.metadata,
      createdAt: authEvents.createdAt,
    })
    .from(authEvents)
    .where(
      and(
        isNotNull(authEvents.userId),
        sql`${authEvents.createdAt} >= ${oneDayAgo.toISOString()}`
      )
    )
    .orderBy(desc(authEvents.createdAt))
    .limit(20);

  // Also read from admin_audit_log if it exists (Phase 1 may have created this)
  // For now, derive feed from auth_events

  // Map auth events to activity items
  const ACTION_VERBS: Record<string, string> = {
    login_success: 'logged in',
    login_failed: 'failed login',
    permission_changed: 'changed permissions for',
    session_created: 'started session',
    totp_enabled: 'enabled 2FA',
    totp_disabled: 'disabled 2FA',
    passkey_registered: 'registered passkey',
    passkey_authenticated: 'authenticated with passkey',
  };

  const adminActionPrefixes = ['clinic_', 'lead_', 'review_', 'blog_', 'settings_'];

  const feedItems = events
    .filter(event => {
      // Include: admin auth events, or events with admin action prefixes
      if (!event.action) return false;
      const action = event.action.toLowerCase();
      return action.startsWith('admin_') ||
             adminActionPrefixes.some(prefix => action.startsWith(prefix)) ||
             Object.keys(ACTION_VERBS).includes(action);
    })
    .slice(0, 10)
    .map((event, index) => {
      const action = event.action || 'performed action';
      const verb = ACTION_VERBS[action] || action.replace(/_/g, ' ');
      const actorId = event.userId || 'unknown';
      const subject = event.metadata && typeof event.metadata === 'object'
        ? ((event.metadata as Record<string, unknown>).target as string || event.action)
        : event.action;

      return {
        id: `${event.id}-${index}`,
        actorName: `Admin ${actorId.slice(0, 8)}`,
        actorInitials: 'A', // In production, would join to users table for real name
        action: action.includes('added') ? 'added' :
                action.includes('removed') ? 'removed' :
                action.includes('edited') || action.includes('updated') ? 'edited' :
                action.includes('approved') || action.includes('verified') ? 'approved' :
                action.includes('rejected') ? 'rejected' :
                action.includes('export') ? 'exported' : 'edited',
        subject: subject || 'item',
        timestamp: event.createdAt?.toISOString() || new Date().toISOString(),
      };
    });

  return json({ items: feedItems }, {
    headers: {
      'Cache-Control': 'no-store',
    },
  });
};
```

```typescript
import { useState, useEffect } from 'react';

interface ActivityItem {
  id: string;
  actorName: string;
  actorInitials: string;
  action: 'added' | 'removed' | 'edited' | 'approved' | 'rejected' | 'exported';
  subject: string;
  timestamp: string;
}

interface AdminActivityFeedProps {
  maxItems?: number;
  className?: string;
}

const ACTION_COLORS: Record<ActivityItem['action'], string> = {
  added: 'text-emerald-600',
  removed: 'text-red-600',
  edited: 'text-violet-600',
  approved: 'text-emerald-600',
  rejected: 'text-red-600',
  exported: 'text-blue-600',
};

function formatRelativeTime(timestamp: string): string {
  const now = new Date();
  const then = new Date(timestamp);
  const diffMs = now.getTime() - then.getTime();
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMins / 60);
  const diffDays = Math.floor(diffHours / 24);

  if (diffMins < 1) return 'just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  if (diffHours < 24) return `${diffHours}h ago`;
  return `${diffDays}d ago`;
}

function getInitialsColor(name: string): string {
  // Consistent color based on name hash
  const colors = [
    'bg-violet-100 text-violet-700',
    'bg-emerald-100 text-emerald-700',
    'bg-amber-100 text-amber-700',
    'bg-blue-100 text-blue-700',
    'bg-rose-100 text-rose-700',
  ];
  let hash = 0;
  for (let i = 0; i < name.length; i++) {
    hash = name.charCodeAt(i) + ((hash << 5) - hash);
  }
  return colors[Math.abs(hash) % colors.length];
}

export default function AdminActivityFeed({ maxItems = 10, className = '' }: AdminActivityFeedProps) {
  const [items, setItems] = useState<ActivityItem[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchFeed = async () => {
    try {
      const res = await fetch('/api/admin/activity-feed');
      if (!res.ok) throw new Error('Failed to load');
      const data = await res.json();
      setItems((data.items || []).slice(0, maxItems));
      setError(null);
    } catch (err) {
      setError('Could not load activity');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchFeed();
    // Poll every 30 seconds
    const interval = setInterval(fetchFeed, 30000);
    return () => clearInterval(interval);
  }, []);

  if (loading) {
    return (
      <div className={`space-y-3 ${className}`}>
        <h3 className="text-xs font-semibold uppercase tracking-wide text-gray-500 px-1">
          Recent Activity
        </h3>
        <div className="space-y-3">
          {[1, 2, 3].map(i => (
            <div key={i} className="animate-pulse flex items-start gap-3">
              <div className="w-8 h-8 rounded-full bg-gray-200 shrink-0" />
              <div className="flex-1 space-y-1.5">
                <div className="h-3 bg-gray-200 rounded w-3/4" />
                <div className="h-2 bg-gray-100 rounded w-1/3" />
              </div>
            </div>
          ))}
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div className={`space-y-2 ${className}`}>
        <h3 className="text-xs font-semibold uppercase tracking-wide text-gray-500 px-1">
          Recent Activity
        </h3>
        <p className="text-xs text-gray-400 px-1">{error}</p>
      </div>
    );
  }

  if (items.length === 0) {
    return (
      <div className={`space-y-2 ${className}`}>
        <h3 className="text-xs font-semibold uppercase tracking-wide text-gray-500 px-1">
          Recent Activity
        </h3>
        <p className="text-xs text-gray-400 px-1 py-2 text-center">No recent activity</p>
      </div>
    );
  }

  return (
    <div className={`space-y-3 ${className}`}>
      <h3 className="text-xs font-semibold uppercase tracking-wide text-gray-500 px-1">
        Recent Activity
      </h3>
      <div className="space-y-3">
        {items.map((item, index) => (
          <div
            key={item.id}
            className="flex items-start gap-3 group animate-in fade-in slide-in-from-top-1"
            style={{ animationDelay: `${index * 30}ms` }}
          >
            {/* Avatar */}
            <div className={`w-8 h-8 rounded-full shrink-0 flex items-center justify-center text-xs font-semibold ${getInitialsColor(item.actorName)}`}>
              {item.actorInitials}
            </div>

            {/* Content */}
            <div className="flex-1 min-w-0">
              <p className="text-xs text-gray-700 leading-relaxed">
                <span className="font-medium text-gray-900">{item.actorName}</span>
                {' '}
                <span className={ACTION_COLORS[item.action]}>{item.action}</span>
                {' '}
                <span className="text-gray-600">{item.subject}</span>
              </p>
              <p className="text-[11px] text-gray-400 mt-0.5">
                {formatRelativeTime(item.timestamp)}
              </p>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}
```

```typescript
'use client';

import { useState, useEffect } from 'react';
import {
  DndContext,
  closestCenter,
  KeyboardSensor,
  PointerSensor,
  useSensor,
  useSensors,
  DragEndEvent,
} from '@dnd-kit/core';
import {
  arrayMove,
  SortableContext,
  sortableKeyboardCoordinates,
  rectSortingStrategy,
} from '@dnd-kit/sortable';

interface WidgetConfig {
  id: string;
  title: string;
  visible: boolean;
  order: number;
}

interface AdminDashboardGridProps {
  widgets: WidgetConfig[];
  onReorder: (widgets: WidgetConfig[]) => void;
  onRemove: (widgetId: string) => void;
  onAdd: (widgetId: string) => void;
  renderWidget: (widget: WidgetConfig) => React.ReactNode;
  hiddenWidgets: WidgetConfig[];
}

function SortableWidget({
  widget,
  onRemove,
  children,
}: {
  widget: WidgetConfig;
  onRemove: (id: string) => void;
  children: React.ReactNode;
}) {
  const [isClient, setIsClient] = useState(false);

  useEffect(() => {
    setIsClient(true);
  }, []);

  return (
    <div className="relative group">
      {children}
      {/* Drag handle and remove button */}
      {isClient && (
        <div className="absolute top-2 right-2 flex items-center gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
          {/* Drag handle */}
          <div className="cursor-grab active:cursor-grabbing p-1 text-gray-400 hover:text-gray-600">
            <svg className="w-4 h-4" fill="currentColor" viewBox="0 0 20 20">
              <path d="M7 2a2 2 0 1 0 0 4 2 2 0 0 0 0-4zM7 8a2 2 0 1 0 0 4 2 2 0 0 0 0-4zM7 14a2 2 0 1 0 0 4 2 2 0 0 0 0-4zM13 2a2 2 0 1 0 0 4 2 2 0 0 0 0-4zM13 8a2 2 0 1 0 0 4 2 2 0 0 0 0-4zM13 14a2 2 0 1 0 0 4 2 2 0 0 0 0-4z" />
            </svg>
          </div>
          {/* Remove button */}
          <button
            onClick={() => onRemove(widget.id)}
            className="p-1 text-gray-400 hover:text-red-600 transition-colors"
            aria-label={`Remove ${widget.title}`}
          >
            <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
            </svg>
          </button>
        </div>
      )}
    </div>
  );
}

export default function AdminDashboardGrid({
  widgets,
  onReorder,
  onRemove,
  onAdd,
  renderWidget,
  hiddenWidgets,
}: AdminDashboardGridProps) {
  const [isClient, setIsClient] = useState(false);

  useEffect(() => {
    setIsClient(true);
  }, []);

  // Sort widgets by order
  const sortedWidgets = [...widgets].sort((a, b) => a.order - b.order);

  const sensors = useSensors(
    useSensor(PointerSensor, {
      activationConstraint: {
        distance: 8,
      },
    }),
    useSensor(KeyboardSensor, {
      coordinateGetter: sortableKeyboardCoordinates,
    })
  );

  function handleDragEnd(event: DragEndEvent) {
    const { active, over } = event;
    if (!over || active.id === over.id) return;

    const oldIndex = sortedWidgets.findIndex(w => w.id === active.id);
    const newIndex = sortedWidgets.findIndex(w => w.id === over.id);
    const reordered = arrayMove(sortedWidgets, oldIndex, newIndex).map((w, i) => ({
      ...w,
      order: i,
    }));
    onReorder(reordered);
  }

  if (!isClient) {
    // SSR: render grid without DnD
    return (
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        {sortedWidgets.map(widget => (
          <SortableWidget key={widget.id} widget={widget} onRemove={onRemove}>
            {renderWidget(widget)}
          </SortableWidget>
        ))}
      </div>
    );
  }

  return (
    <DndContext
      sensors={sensors}
      collisionDetection={closestCenter}
      onDragEnd={handleDragEnd}
    >
      <SortableContext
        items={sortedWidgets.map(w => w.id)}
        strategy={rectSortingStrategy}
      >
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
          {sortedWidgets.map(widget => (
            <SortableWidget key={widget.id} widget={widget} onRemove={onRemove}>
              {renderWidget(widget)}
            </SortableWidget>
          ))}
        </div>
      </SortableContext>
    </DndContext>
  );
}
```
