```typescript
'use client';
import { useState, useCallback } from 'react';
import { useMarkdownEditor } from '../../hooks/useMarkdownEditor';

interface Props {
  post?: { id?: string; title: string; content: string; status: string; category: string; tags: string[] };
  onSave: (post: any) => void;
  onCancel: () => void;
}

export default function AdminBlogEditor({ post, onSave, onCancel }: Props) {
  const [title, setTitle] = useState(post?.title || '');
  const [content, setContent] = useState(post?.content || '');
  const [category, setCategory] = useState(post?.category || 'research');
  const [tags, setTags] = useState(post?.tags?.join(', ') || '');
  const [saving, setSaving] = useState(false);
  const { preview } = useMarkdownEditor();

  const handleSave = async (publish = false) => {
    setSaving(true);
    try {
      await onSave({
        id: post?.id,
        title,
        content,
        category,
        tags: tags.split(',').map(t => t.trim()).filter(Boolean),
        status: publish ? 'published' : (post?.status || 'draft'),
      });
    } finally {
      setSaving(false);
    }
  };

  return (
    <div className="grid grid-cols-2 gap-6 h-[calc(100vh-200px)]">
      {/* Editor pane */}
      <div className="flex flex-col">
        <input value={title} onChange={e => setTitle(e.target.value)}
          placeholder="Post title..." className="text-2xl font-semibold border-b p-4" />
        <div className="flex-1 p-4">
          <textarea value={content} onChange={e => setContent(e.target.value)}
            className="w-full h-full resize-none font-mono" placeholder="Write in Markdown..." />
        </div>
        <div className="flex gap-2 p-4 border-t">
          <button onClick={() => handleSave(false)} disabled={saving}
            className="px-4 py-2 bg-gray-200 rounded">Save Draft</button>
          <button onClick={() => handleSave(true)} disabled={saving}
            className="px-4 py-2 bg-violet-600 text-white rounded">Publish</button>
          <button onClick={onCancel} className="px-4 py-2 border rounded">Cancel</button>
        </div>
      </div>
      {/* Preview pane */}
      <div className="border-l p-4 overflow-auto">
        <h1 className="text-2xl font-semibold mb-4">{title || 'Untitled'}</h1>
        <div className="prose" dangerouslySetInnerHTML={{ __html: preview(content) }} />
      </div>
    </div>
  );
}
```
