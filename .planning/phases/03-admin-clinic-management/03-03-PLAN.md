```typescript
import { db } from '../../../../../db';
import { clinicBadges } from '../../../../../db/schema';
import { eq, and } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

const BADGE_NAMES: Record<string, string> = {
  verified: 'Verified',
  top_rated: 'Top Rated',
  new: 'New',
  accepting_new_patients: 'Accepting New Patients',
  featured: 'Featured',
};

export const PUT: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin', 'editor')) return json({ error: 'Forbidden' }, 403);

  try {
    const { badgeTypes } = await request.json() as { badgeTypes: string[] };
    const clinicId = params.id!;

    // Remove all existing badges for this clinic
    await db.delete(clinicBadges).where(eq(clinicBadges.clinicId, clinicId));

    // Insert new badges
    const now = new Date();
    for (const bt of badgeTypes) {
      await db.insert(clinicBadges).values({
        clinicId,
        badgeType: bt,
        badgeName: BADGE_NAMES[bt] || bt,
      });
    }

    return json({ success: true, count: badgeTypes.length });
  } catch (err) {
    console.error('Badge update error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
'use client';
import { useState, useCallback, useRef } from 'react';

interface PlaceSuggestion {
  description: string;
  place_id: string;
  types: string[];
}

export function usePlacesAutocomplete() {
  const [suggestions, setSuggestions] = useState<PlaceSuggestion[]>([]);
  const [loading, setLoading] = useState(false);
  const timerRef = useRef<ReturnType<typeof setTimeout>>();

  const fetchSuggestions = useCallback((input: string) => {
    clearTimeout(timerRef.current);
    if (!input || input.length < 3) {
      setSuggestions([]);
      return;
    }

    timerRef.current = setTimeout(async () => {
      const apiKey = (window as any).__GOOGLE_PLACES_KEY__;
      if (!apiKey) {
        setSuggestions([]);
        return;
      }

      setLoading(true);
      try {
        const res = await fetch(
          `https://maps.googleapis.com/maps/api/place/autocomplete/json?input=${encodeURIComponent(input)}&key=${apiKey}&types=address`
        );
        const data = await res.json();
        setSuggestions(data.predictions || []);
      } catch {
        setSuggestions([]);
      } finally {
        setLoading(false);
      }
    }, 300);
  }, []);

  const getPlaceDetails = useCallback(async (placeId: string): Promise<Record<string, string>> => {
    const apiKey = (window as any).__GOOGLE_PLACES_KEY__;
    if (!apiKey) return {};

    const res = await fetch(
      `https://maps.googleapis.com/maps/api/place/details/json?place_id=${placeId}&key=${apiKey}`
    );
    const data = await res.json();
    const result = data.result?.address_components || [];

    const parts: Record<string, string> = {};
    for (const comp of result) {
      if (comp.types.includes('street_number')) parts.street_number = comp.long_name;
      if (comp.types.includes('route')) parts.route = comp.long_name;
      if (comp.types.includes('locality')) parts.city = comp.long_name;
      if (comp.types.includes('administrative_area_level_1')) parts.state = comp.long_name;
      if (comp.types.includes('postal_code')) parts.zip = comp.long_name;
    }

    return {
      address: [parts.street_number, parts.route].filter(Boolean).join(' '),
      city: parts.city || '',
      state: parts.state || '',
      zip: parts.zip || '',
    };
  }, []);

  return { suggestions, loading, fetchSuggestions, getPlaceDetails };
}
```

```typescript
'use client';
import { useState, useRef } from 'react';
import { usePlacesAutocomplete } from '../../hooks/usePlacesAutocomplete';

interface Props {
  value: string;
  onChange: (value: string) => void;
  onPlaceSelect?: (details: { address: string; city: string; state: string; zip: string }) => void;
}

export default function AdminAddressAutocomplete({ value, onChange, onPlaceSelect }: Props) {
  const [focused, setFocused] = useState(false);
  const { suggestions, loading, fetchSuggestions, getPlaceDetails } = usePlacesAutocomplete();
  const [selectedIndex, setSelectedIndex] = useState(-1);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    onChange(e.target.value);
    fetchSuggestions(e.target.value);
    setSelectedIndex(-1);
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (!suggestions.length) return;
    if (e.key === 'ArrowDown') setSelectedIndex(i => Math.min(i + 1, suggestions.length - 1));
    if (e.key === 'ArrowUp') setSelectedIndex(i => Math.max(i - 1, -1));
    if (e.key === 'Enter' && selectedIndex >= 0) {
      handleSelect(suggestions[selectedIndex]);
    }
    if (e.key === 'Escape') setSuggestions([]);
  };

  const handleSelect = async (place: { place_id: string; description: string }) => {
    onChange(place.description);
    const details = await getPlaceDetails(place.place_id);
    if (onPlaceSelect) onPlaceSelect(details);
    setSuggestions([]);
  };

  const showDropdown = focused && suggestions.length > 0;

  return (
    <div className="relative">
      <input
        type="text"
        value={value}
        onChange={handleChange}
        onKeyDown={handleKeyDown}
        onFocus={() => setFocused(true)}
        onBlur={() => setTimeout(() => setFocused(false), 200)}
        placeholder="Start typing address..."
        className="w-full px-3 py-2 text-sm border border-gray-200 rounded-lg focus:ring-2 focus:ring-violet-500 outline-none"
      />
      {loading && <span className="absolute right-3 top-2.5 text-xs text-gray-400">...</span>}
      {showDropdown && (
        <ul className="absolute z-10 w-full bg-white border border-gray-200 rounded-lg shadow-lg mt-1 text-sm">
          {suggestions.map((s, i) => (
            <li
              key={s.place_id}
              onMouseDown={() => handleSelect(s)}
              className={`px-3 py-2 cursor-pointer hover:bg-gray-50 ${i === selectedIndex ? 'bg-violet-50' : ''}`}
            >
              {s.description}
            </li>
          ))}
        </ul>
      )}
      {!suggestions.length && !loading && value && (
        <p className="text-xs text-gray-400 mt-1">Configure GOOGLE_PLACES_API_KEY for autocomplete</p>
      )}
    </div>
  );
}
```

```typescript
'use client';
import { useState, useMemo } from 'react';
import INSURANCE_DATA from '../../data/insurance-providers.json';

interface Props {
  selected: string[];
  onChange: (insurances: string[]) => void;
}

export default function AdminInsuranceAutocomplete({ selected, onChange }: Props) {
  const [input, setInput] = useState('');
  const [focused, setFocused] = useState(false);

  const filtered = useMemo(() => {
    if (!input) return INSURANCE_DATA.map(i => i.name);
    const q = input.toLowerCase();
    return INSURANCE_DATA
      .filter(i => i.name.toLowerCase().includes(q) || i.aliases.some(a => a.includes(q)))
      .map(i => i.name);
  }, [input]);

  const add = (name: string) => {
    if (!selected.includes(name)) {
      onChange([...selected, name]);
    }
    setInput('');
  };

  const remove = (name: string) => {
    onChange(selected.filter(s => s !== name));
  };

  const showDropdown = focused && input.length > 0;

  return (
    <div>
      {selected.length > 0 && (
        <div className="flex flex-wrap gap-1.5 mb-2">
          {selected.map(s => (
            <span key={s} className="inline-flex items-center gap-1 px-2 py-1 bg-violet-100 text-violet-700 text-xs rounded-full">

            </span>
          ))}
        </div>
      )}
      <div className="relative">
        <input
          type="text"
          value={input}
          onChange={e => setInput(e.target.value)}
          onFocus={() => setFocused(true)}
          onBlur={() => setTimeout(() => setFocused(false), 200)}
          onKeyDown={e => {
            if (e.key === 'Enter' && filtered.length > 0) {
              e.preventDefault();
              add(filtered[0]);

          className="w-full px-3 py-2 text-sm border border-gray-200 rounded-lg focus:ring-2 focus:ring-violet-500 outline-none"
```

```typescript
'use client';
import { useState, useMemo } from 'react';

const COMMON_TAGS = [
  'TMS Therapy', 'Depression Treatment', 'Anxiety Treatment', 'OCD',
  'PTSD', 'Neuroplasticity', 'Medication-Free', 'Insurance Accepted',
  'Telehealth', 'In-Person', 'Outpatient', 'Intensive Outpatient',
  'FDA Cleared', 'Deep TMS', 'Fast TMS', 'Standard TMS',
];

interface Props {
  selected: string[];
  onChange: (tags: string[]) => void;
}

export default function AdminTagManager({ selected, onChange }: Props) {
  const [input, setInput] = useState('');
  const [focused, setFocused] = useState(false);

  const filtered = useMemo(() => {
    if (!input) return COMMON_TAGS.filter(t => !selected.includes(t));
    const q = input.toLowerCase();
    return COMMON_TAGS
      .filter(t => t.toLowerCase().includes(q) && !selected.includes(t));
  }, [input, selected]);

  const add = (tag: string) => {
    if (!selected.includes(tag)) {
      onChange([...selected, tag]);
    }
    setInput('');
  };

  const remove = (tag: string) => {
    onChange(selected.filter(t => t !== tag));
  };

  return (
    <div>
      {selected.length > 0 && (
        <div className="flex flex-wrap gap-1.5 mb-3">
          {selected.map(tag => (
            <span key={tag} className="inline-flex items-center gap-1 px-2.5 py-1 bg-violet-100 text-violet-700 text-xs rounded-full font-medium">
              {tag}
              <button onClick={() => remove(tag)} className="hover:text-violet-900 font-bold leading-none">×</button>
            </span>
          ))}
        </div>
      )}
      <div className="relative">
        <input
          type="text"
          value={input}
          onChange={e => setInput(e.target.value)}
          onFocus={() => setFocused(true)}
          onBlur={() => setTimeout(() => setFocused(false), 200)}
          onKeyDown={e => {
            if (e.key === 'Enter' && input.trim()) {
              e.preventDefault();
              add(input.trim());
            }
            if (e.key === 'Backspace' && !input && selected.length > 0) {
              remove(selected[selected.length - 1]);
            }
          }}
          placeholder="Add tags (press Enter)..."
          className="w-full px-3 py-2 text-sm border border-gray-200 rounded-lg focus:ring-2 focus:ring-violet-500 outline-none"
        />
        {focused && (input.length > 0 || filtered.length > 0) && (
          <div className="absolute z-10 w-full bg-white border border-gray-200 rounded-lg shadow-lg mt-1 text-sm">
            {input.trim() && !COMMON_TAGS.includes(input.trim()) && (
              <div
                onMouseDown={() => add(input.trim())}
                className="px-3 py-2 cursor-pointer hover:bg-violet-50 text-violet-600 border-b border-gray-100 font-medium"
              >
                + Add "{input.trim()}"
              </div>
            )}
            {filtered.slice(0, 8).map(tag => (
              <div
                key={tag}
                onMouseDown={() => add(tag)}
                className="px-3 py-2 cursor-pointer hover:bg-gray-50 text-gray-600"
              >
                {tag}
              </div>
            ))}
          </div>
        )}
      </div>
      <p className="text-xs text-gray-400 mt-1">{selected.length} tags assigned</p>
    </div>
  );
}
```
