```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../db';
import { clinics } from '../../../../db/schema';
import { getSessionFromRequest, hasRole } from '../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const GET: APIRoute = async ({ request }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin', 'editor')) return json({ error: 'Forbidden' }, 403);

  try {
    // Find potential duplicates: same name (normalized) + same city
    const all = await db.select({
      id: clinics.id,
      name: clinics.name,
      city: clinics.city,
      state: clinics.state,
      address: clinics.address,
      phone: clinics.phone,
    }).from(clinics).where(sql`${clinics.deletedAt} IS NULL`);

    // Simple duplicate detection: normalize name and group
    function normalize(s: string) {
      return s.toLowerCase().replace(/[^a-z0-9]/g, '').trim();
    }

    const groups: Record<string, typeof all> = {};
    for (const c of all) {
      const key = `${normalize(c.name)}|${c.city.toLowerCase()}`;
      if (!groups[key]) groups[key] = [];
      groups[key].push(c);
    }

    const duplicates = Object.values(groups)
      .filter(g => g.length > 1)
      .slice(0, 50);

    return json({ duplicates });
  } catch (err) {
    console.error('Duplicates error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../../db';
import { clinics } from '../../../../../db/schema';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const POST: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin')) return json({ error: 'Forbidden' }, 403);

  try {
    const { targetId, sourceId, keepFields } = await request.json() as {
      targetId: string;
      sourceId: string;
      keepFields: Record<string, 'source' | 'target'>;
    };

    const source = await db.select().from(clinics).where(eq(clinics.id, sourceId)).limit(1);
    const target = await db.select().from(clinics).where(eq(clinics.id, targetId)).limit(1);

    if (!source[0] || !target[0]) {
      return json({ error: 'Clinic not found' }, 404);
    }

    // Build merged values
    const merged: Partial<typeof clinics.$inferInsert> = {};
    const src = source[0];
    const tgt = target[0];

    for (const [field, choice] of Object.entries(keepFields)) {
      merged[field as keyof typeof src] = choice === 'source'
        ? src[field as keyof typeof src]
        : tgt[field as keyof typeof tgt];
    }

    // Soft delete source clinic
    await db.update(clinics)
      .set({ deletedAt: new Date(), slug: `${src.slug}-merged-${Date.now().toString(36)}` })
      .where(eq(clinics.id, sourceId));

    // Update target
    await db.update(clinics)
      .set({ ...merged, updatedAt: new Date() })
      .where(eq(clinics.id, targetId));

    return json({ success: true });
  } catch (err) {
    console.error('Merge error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
interface Props {

  currentNPI: string;
  currentStatus: 'verified' | 'flagged' | 'pending';
}

export default function AdminNPIValidationPanel({ clinicId, currentNPI, currentStatus }: Props) {
  const [npi, setNpi] = useState(currentNPI || '');
  const [validating, setValidating] = useState(false);

    valid: boolean;
    providerName?: string;
    error?: string;
  } | null>(null);

    if (!npi || npi.length !== 10) {
      setResult({ valid: false, error: 'NPI must be exactly 10 digits' });
      return;
    }
    setValidating(true);
    try {
      const res = await fetch(`/api/admin/clinics/${clinicId}/verify`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ npi }),
      });
      const data = await res.json();
      setResult(data);
    } catch {
      setResult({ valid: false, error: 'Validation request failed' });
    } finally {
      setValidating(false);
    }
  };

  const statusColors: Record<string, string> = {
    verified: 'bg-emerald-100 text-emerald-700',
    flagged: 'bg-red-100 text-red-700',
    pending: 'bg-amber-100 text-amber-700',
  };

  return (
    <div className="border border-gray-100 rounded-lg p-4 bg-gray-50">
      <div className="flex items-center justify-between mb-3">
        <h4 className="text-sm font-semibold text-gray-900">NPI Verification</h4>
        <span className={`px-2 py-0.5 text-xs rounded-full ${statusColors[currentStatus] || 'bg-gray-100'}`}>
          {currentStatus}
        </span>
      </div>

        <input
          type="text"
          value={npi}

        </div>
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../../db';
import { clinics } from '../../../../../db/schema';
import { eq } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

function validateNPIFormat(npi: string): boolean {
  return /^\d{10}$/.test(npi) && ['1', '2'].includes(npi[0]);
}

export const POST: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin', 'editor')) return json({ error: 'Forbidden' }, 403);

  const id = params.id;
  try {
    const { npi } = await request.json() as { npi: string };

    if (!validateNPIFormat(npi)) {
      return json({ valid: false, error: 'Invalid NPI format — must be 10 digits starting with 1 or 2' });
    }

    // Call NPI Registry API
    const res = await fetch(
      `https://npiregistry.cms.hhs.gov/api/?number=${npi}&version=2.1`,
      { headers: { 'Accept': 'application/json' } }
    );

    if (!res.ok) {
      return json({ valid: false, error: 'NPI Registry API unavailable' });
    }

    const data = await res.json() as { results?: Array<{ basic?: { first_name?: string; last_name?: string } }> };
    const result = data.results?.[0];

    if (!result) {
      // NPI not found in registry — still save but flag
      await db.update(clinics)
        .set({ npi, verified: false, updatedAt: new Date() })
        .where(eq(clinics.id, id));
      return json({ valid: false, error: 'NPI not found in CMS registry — flagged for manual review' });
    }

    const providerName = [result.basic?.first_name, result.basic?.last_name].filter(Boolean).join(' ');

    // Auto-verify if NPI found
    await db.update(clinics)
      .set({ npi, verified: true, updatedAt: new Date() })
      .where(eq(clinics.id, id));

    return json({ valid: true, providerName });
  } catch (err) {
    console.error('NPI verify error:', err);
    return json({ valid: false, error: 'Verification failed' });
  }
};
```

```typescript
    return json({
      clinics: all.filter(c => !c.npi || c.npi.trim() === ''),
    });
  }
};
```

```typescript
import { useState, useEffect } from 'react';

interface Claim {
  id: string;
  clinicId: string;
  clinicName: string;
  email: string;
  status: 'pending' | 'approved' | 'rejected' | 'expired';
  createdAt: string;
}

interface Props {
  onClose: () => void;
}

export default function AdminTakeoverRequestModal({ onClose }: Props) {
  const [claims, setClaims] = useState<Claim[]>([]);
  const [loading, setLoading] = useState(true);
  const [processing, setProcessing] = useState<string | null>(null);
  const [rejectReason, setRejectReason] = useState('');

  useEffect(() => {
    fetch('/api/admin/clinics/claims')
      .then(r => r.json())
      .then(data => {
        setClaims(data.claims || []);
        setLoading(false);
      });
  }, []);

  const handleAction = async (claimId: string, action: 'approve' | 'reject') => {
    if (action === 'reject' && !rejectReason.trim()) {
      alert('Please provide a rejection reason');
      return;
    }
    setProcessing(claimId);
    try {
      const res = await fetch('/api/admin/clinics/claims', {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ claimId, action, reason: rejectReason }),
      });
      if (res.ok) {
        setClaims(prev => prev.filter(c => c.id !== claimId));
      }
    } finally {
      setProcessing(null);
    }
  };

  const pending = claims.filter(c => c.status === 'pending');
  const resolved = claims.filter(c => c.status !== 'pending').slice(0, 20);

  return (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-4">

  }
```

```typescript
export const PUT: APIRoute = async ({ request }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin')) return json({ error: 'Forbidden' }, 403);

  try {
    const { claimId, action, reason } = await request.json() as {
      claimId: string;
      action: 'approve' | 'reject';
      reason?: string;
    };

    const newStatus = action === 'approve' ? 'approved' : 'rejected';

    await db.update(clinicClaims)
      .set({
        status: newStatus,
        verifiedAt: action === 'approve' ? new Date() : undefined,
        metadata: reason ? { rejectionReason: reason } : undefined,
      })
      .where(eq(clinicClaims.id, claimId));

    return json({ success: true, status: newStatus });
  } catch (err) {
    console.error('Claims update error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../db';
import { clinicClaims, clinics } from '../../../../db/schema';
import { eq } from 'drizzle-orm';
import { randomBytes } from 'crypto';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const POST: APIRoute = async ({ request, params }) => {
  const id = params.id;
  try {
    const { email } = await request.json() as { email: string };

    if (!email || !email.includes('@')) {
      return json({ error: 'Valid email required' }, 400);
    }

    // Check clinic exists
    const clinic = await db.select({ id: clinics.id, name: clinics.name })
      .from(clinics).where(eq(clinics.id, id)).limit(1);
    if (!clinic[0]) return json({ error: 'Clinic not found' }, 404);

    const token = randomBytes(32).toString('hex');
    const expiresAt = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000); // 7 days

    await db.insert(clinicClaims).values({
      clinicId: id,
      email,
      verificationToken: token,
      expiresAt,
    });

    return json({ success: true, message: `Claim request submitted for ${clinic[0].name}` });
  } catch (err) {
    console.error('Claim error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```
