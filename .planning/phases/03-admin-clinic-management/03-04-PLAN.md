```typescript
  city: string;
  state: string;

  clinic: Clinic;
  onClose: () => void;
  onDeleted: () => void;

  const [targetId, setTargetId] = useState('');
  const [redirectEnabled, setRedirectEnabled] = useState(false);
  const [deleting, setDeleting] = useState(false);

    fetch('/api/admin/clinics?limit=100')
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../../db';
import { clinics, slugRedirects } from '../../../../../db/schema';
import { eq } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const DELETE: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin')) return json({ error: 'Forbidden' }, 403);

  const id = params.id;
  try {
    const body = await request.json().catch(() => ({}));
    const { redirectTo } = body as { redirectTo: string | null };

    // Soft delete
    await db.update(clinics)
      .set({ deletedAt: new Date(), updatedAt: new Date() })
      .where(eq(clinics.id, id));

    // Create redirect if requested
    if (redirectTo) {
      const clinic = await db.select({ slug: clinics.slug }).from(clinics).where(eq(clinics.id, id)).limit(1);
      if (clinic[0]) {
        await db.insert(slugRedirects).values({
          oldSlug: clinic[0].slug,
          newClinicId: redirectTo,
          createdBy: session.userId,
        }).onConflictDoNothing();
      }
    }

    return json({ success: true });
  } catch (err) {
    console.error('Delete clinic error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
  'Business permanently closed',
  'Does not meet TMS/treatment directory criteria',
  'Other (specify below)',
];

const PAUSE_REASONS = [
  'Temporarily not accepting new patients',
  'Business temporarily closed',
  'Awaiting verification documents',
  'Seasonal closure',
  'Provider sabbatical or leave',
  'Other (specify below)',
];

interface Props {
  clinic: { id: string; name: string };
  action: 'reject' | 'pause';
  onClose: () => void;
  onComplete: (reason: string) => void;
}

export default function AdminStatusReasonModal({ clinic, action, onClose, onComplete }: Props) {
  const [reason, setReason] = useState('');
  const [details, setDetails] = useState('');
  const [submitting, setSubmitting] = useState(false);

          action,
          reason: details ? `${reason}: ${details}` : reason,
        }),
      });
      if (res.ok) {
        onComplete(reason);
        onClose();
      }
    } finally {
      setSubmitting(false);
    }
  };

  return (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-xl shadow-2xl w-full max-w-md">
        <div className="flex items-center gap-3 px-6 py-4 border-b">
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../../db';
import { clinics } from '../../../../../db/schema';
import { eq } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const PUT: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin', 'editor')) return json({ error: 'Forbidden' }, 403);

  const id = params.id;
  try {
    const { action, reason } = await request.json() as { action: string; reason: string };

    if (!reason) return json({ error: 'Reason is required' }, 400);

    if (action === 'reject') {
      await db.update(clinics)
        .set({ verified: false, statusReason: reason, updatedAt: new Date() })
        .where(eq(clinics.id, id));
    } else if (action === 'pause') {
      await db.update(clinics)
        .set({ statusReason: reason, updatedAt: new Date() })
        .where(eq(clinics.id, id));
    }

    return json({ success: true });
  } catch (err) {
    console.error('Status update error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../../../db';
import { clinics } from '../../../../../db/schema';
import { eq } from 'drizzle-orm';
import { getSessionFromRequest, hasRole } from '../../../../../utils/auth';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), {
    status,
    headers: { 'Content-Type': 'application/json' },
  });

export const POST: APIRoute = async ({ request, params }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin', 'editor')) return json({ error: 'Forbidden' }, 403);

  const sourceId = params.id;
  try {
    const { name, slug } = await request.json() as { name: string; slug: string };

    if (!name || !slug) return json({ error: 'Name and slug are required' }, 400);

    // Fetch source clinic
    const [source] = await db.select().from(clinics).where(eq(clinics.id, sourceId)).limit(1);
    if (!source) return json({ error: 'Source clinic not found' }, 404);

    // Insert clone
    const [cloned] = await db.insert(clinics).values({
      name,
      slug,
      address: source.address,
      city: source.city,
      state: source.state,
      zip: source.zip,
      country: source.country,
      lat: source.lat,
      lng: source.lng,
      phone: source.phone,
      website: source.website,
      email: null,
      description: source.description,
      descriptionLong: source.descriptionLong,
      machines: source.machines,
      specialties: source.specialties,
      insurances: source.insurances,
      openingHours: source.openingHours,
      verified: false,
      isFeatured: false,
      ratingAvg: '0',
      reviewCount: 0,
      ownerUserId: null,
      ownerEmail: null,
      ownerName: null,
      createdBy: { name: 'Clone', source: 'admin', clonedFrom: sourceId },
    }).returning({ id: clinics.id });

    return json({ success: true, id: cloned.id });
  } catch (err: any) {
    if (err?.code === '23505') {
      return json({ error: 'Slug already exists — please choose a different one' }, 400);
    }
    console.error('Clone error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```
