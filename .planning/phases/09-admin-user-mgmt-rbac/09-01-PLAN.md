```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../db';
import { users, magicTokens } from '../../../db/schema';
import { getSessionFromRequest, hasRole, createMagicToken } from '../../../utils/auth';
import { eq } from 'drizzle-orm';
import { sendTransactionalEmail } from '../../../utils/email';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), { status, headers: { 'Content-Type': 'application/json' } });

export const POST: APIRoute = async ({ request }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin')) return json({ error: 'Forbidden' }, 403);

  try {
    const body = await request.json();
    const { email, name, role = 'editor', permissions = [] } = body;

    if (!email || !name) {
      return json({ error: 'email and name are required' }, 400);
    }

    // Check if user already exists
    const [existing] = await db.select().from(users).where(eq(users.email, email.toLowerCase())).limit(1);
    if (existing) {
      return json({ error: 'User with this email already exists' }, 400);
    }

    // Create invite token (purpose: 'invite')
    const token = await createMagicToken(email, 'invite');

    // Store invite metadata in magicTokens metadata field
    await db.update(magicTokens)
      .set({ metadata: { name, role, permissions, invitedBy: session.userId } })
      .where(eq(magicTokens.token, require('crypto').createHash('sha256').update(token).digest('hex')));

    // Send invite email
    const inviteUrl = `${process.env.SITE_URL || 'https://tmslist.com'}/admin/accept-invite?token=${token}`;
    await sendTransactionalEmail({
      to: email,
      subject: 'You are invited to join TMS List Admin',
      html: `
        <h1>You've been invited to TMS List Admin</h1>
        <p>Hi ${name},</p>
        <p>You've been invited to join the TMS List admin panel as a ${role}.</p>
        <p><a href="${inviteUrl}">Click here to accept your invitation and set your password</a></p>
        <p>This link expires in 7 days.</p>
      `,
    });

    return json({ success: true, message: 'Invitation sent' }, 201);
  } catch (err) {
    console.error('Invite error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

Note: The `createMagicToken` function from auth.ts stores token hash, so we need to find the token by email to update metadata. Better approach is to pass metadata directly to a new `createInviteToken` function that stores metadata atomically.

Create a helper in auth.ts:

export async function createInviteToken(email: string, metadata: { name: string; role: string; permissions: string[]; invitedBy: string }) {
  const token = randomBytes(32).toString('hex');
  const tokenHash = createHash('sha256').update(token).digest('hex');
  const expiresAt = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000); // 7 days

  await db.insert(magicTokens).values({
    email: email.toLowerCase(),
    token: tokenHash,
    purpose: 'invite',
    expiresAt,
    metadata,
  });
  return token;
}

```typescript
import type { APIRoute } from 'astro';
import { db } from '../../../db';
import { users, magicTokens } from '../../../db/schema';
import { createSession } from '../../../utils/auth';
import { eq, and, isNull, gt } from 'drizzle-orm';
import { createHash } from 'crypto';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), { status, headers: { 'Content-Type': 'application/json' } });

export const POST: APIRoute = async ({ request }) => {
  try {
    const body = await request.json();
    const { token, password } = body;

    if (!token || !password) {
      return json({ error: 'token and password are required' }, 400);
    }

    if (password.length < 8) {
      return json({ error: 'Password must be at least 8 characters' }, 400);
    }

    // Verify invite token
    const tokenHash = createHash('sha256').update(token).digest('hex');
    const [invite] = await db.select().from(magicTokens)
      .where(and(
        eq(magicTokens.token, tokenHash),
        eq(magicTokens.purpose, 'invite'),
        isNull(magicTokens.usedAt),
        gt(magicTokens.expiresAt, new Date()),
      ))
      .limit(1);

    if (!invite) {
      return json({ error: 'Invalid or expired invitation' }, 400);
    }

    // Get invite metadata
    const metadata = invite.metadata as { name?: string; role?: string; permissions?: string[] } || {};

    // Create user
    const { hashPassword } = await import('../../../utils/auth');
    const hashedPassword = await hashPassword(password);

    const [user] = await db.insert(users).values({
      email: invite.email,
      name: metadata.name || 'New User',
      passwordHash: hashedPassword,
      role: metadata.role || 'editor',
      permissions: metadata.permissions || [],
      emailVerified: true,
      onboardingCompletedAt: new Date(),
    }).returning();

    // Mark token as used
    await db.update(magicTokens)
      .set({ usedAt: new Date() })
      .where(eq(magicTokens.id, invite.id));

    // Create session
    const session = await createSession(user.id, 'admin');

    return json({
      success: true,
      user: { id: user.id, email: user.email, name: user.name, role: user.role },
      sessionToken: session.token,
    });
  } catch (err) {
    console.error('Accept invite error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};
```

This endpoint is public (no auth required) since the user doesn't have an account yet.

import type { APIRoute } from 'astro';
import { db } from '../../../db';
import { sessions, users } from '../../../db/schema';
import { getSessionFromRequest, hasRole } from '../../../utils/auth';
import { desc, eq } from 'drizzle-orm';

export const prerender = false;

const json = (data: unknown, status = 200) =>
  new Response(JSON.stringify(data), { status, headers: { 'Content-Type': 'application/json' } });

export const GET: APIRoute = async ({ request }) => {
  const session = getSessionFromRequest(request);
  if (!session) return json({ error: 'Unauthorized' }, 401);
  if (!hasRole(session, 'admin')) return json({ error: 'Forbidden' }, 403);

  try {
    // Get all active sessions with user info
    const activeSessions = await db
      .select({
        id: sessions.id,
        userId: sessions.userId,
        userEmail: users.email,
        userName: users.name,
        userRole: users.role,
        createdAt: sessions.createdAt,
        expiresAt: sessions.expiresAt,
        ipAddress: sessions.ipAddress,
        userAgent: sessions.userAgent,
      })
      .from(sessions)
      .leftJoin(users, eq(sessions.userId, users.id))
      .where(eq(sessions.revokedAt, null))
      .orderBy(desc(sessions.createdAt))
      .limit(100);

    // Also get sessions from magicTokens for pending invites (optional)
    const pendingInvites = await db
      .select({
        id: sessions.id,
        email: sessions.email,
        createdAt: sessions.createdAt,
        expiresAt: sessions.expiresAt,
      })
      .from(sessions)
      .where(eq(sessions.revokedAt, null));

    return json({ sessions: activeSessions, total: activeSessions.length });
  } catch (err) {
    console.error('Sessions GET error:', err);
    return json({ error: 'Internal server error' }, 500);
  }
};

Create `src/middleware/auth.ts` for permission enforcement. This middleware will intercept all `/api/admin/*` requests and enforce RBAC.

```typescript
import type { MiddlewareHandler } from 'astro';
import { getSessionFromRequest, hasRole, hasPermission } from '../utils/auth';

interface PermissionConfig {
  role?: string[];
  permissions?: string[];
}

// Define permission requirements for each admin endpoint pattern
const PERMISSION_MAP: Record<string, PermissionConfig | ((req: Request) => PermissionConfig)> = {
  // Users management
  '/api/admin/users': { role: ['admin'] },
  '/api/admin/users/invite': { role: ['admin'] },
  '/api/admin/sessions': { role: ['admin'] },
  '/api/admin/sessions/': { role: ['admin'] },

  // Clinic management
  '/api/admin/clinics': { role: ['admin', 'editor'] },
  '/api/admin/clinics/': { role: ['admin', 'editor'] },

  // Reviews & Leads
  '/api/admin/reviews': { role: ['admin', 'editor'] },
  '/api/admin/leads': { role: ['admin', 'editor'] },

  // Billing & Revenue
  '/api/admin/billing': { role: ['admin'] },
  '/api/admin/invoicing': { role: ['admin'] },

  // Content management
  '/api/admin/blog': { role: ['admin', 'editor'] },
  '/api/admin/newsletter': { role: ['admin', 'editor'] },

  // Settings
  '/api/admin/settings': { role: ['admin'] },

  // Delete actions require admin
  '/api/admin/.*/delete': { role: ['admin'] },
};

export function requirePermission(permission: string) {
  return async (context: { request: Request }) => {
    const session = getSessionFromRequest(context.request);
    if (!session) {
      return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
    }
    if (!hasPermission(session, permission)) {
      return new Response(JSON.stringify({ error: 'Forbidden' }), { status: 403 });
    }
    return null; // Allow request to proceed
  };
}

export async function checkPermissions(request: Request, path: string): Promise<Response | null> {
  const session = getSessionFromRequest(request);
  if (!session) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
  }

  // Find matching permission config
  for (const [pattern, config] of Object.entries(PERMISSION_MAP)) {
    const regex = new RegExp(pattern.replace(/\//g, '\\/').replace(/\*/g, '.*'));
    if (regex.test(path)) {
      const cfg = typeof config === 'function' ? config(request) : config;

      // Check role
      if (cfg.role && !hasRole(session, ...cfg.role)) {
        return new Response(JSON.stringify({ error: 'Forbidden: insufficient role' }), { status: 403 });
      }

      // Check permissions
      if (cfg.permissions) {
        for (const perm of cfg.permissions) {
          if (!hasPermission(session, perm)) {
            return new Response(JSON.stringify({ error: 'Forbidden: missing permission' }), { status: 403 });
          }
        }
      }

      break;
    }
  }

  return null; // No blocking
}
```

Add `hasPermission` to auth.ts:

export function hasPermission(session: JWTPayload, permission: string): boolean {
  if (hasRole(session, 'admin')) return true; // Admins have all permissions
  const userPerms = (session as any).permissions || [];
  return userPerms.includes(permission) || userPerms.includes('*');
}

// src/pages/api/admin/_guard.ts
import { getSessionFromRequest, hasRole, hasPermission } from '../../../utils/auth';

export function adminGuard(request: Request) {
  const session = getSessionFromRequest(request);
  if (!session) return { error: 'Unauthorized', status: 401 };
  return { session };
}

export function requireRole(...roles: string[]) {
  return (request: Request) => {
    const { session } = adminGuard(request);
    if (!hasRole(session, ...roles)) {
      return { error: 'Forbidden', status: 403 };
    }
    return { session };
  };
}
